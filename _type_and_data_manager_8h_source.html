826  // _dimensionAndSlabLengths works as follows
827  // For example, in an array of Rank 2, there will be 2 DimensionLengths followed by
828  // 2 slabLengths. slabLengths are precalculated in AQSize used for indexing.
829  // For the inner most dimension the slab length is the length of the element.

877  IntIndex GetLength(IntIndex i)

879  VIREO_ASSERT((i >= 0) && (i < Type()->Rank())); // TODO remove, initially I want to catch any of these.
880  if ((i >= 0) && (i < Type()->Rank())) {

887  // Total Length (product of all dimension lengths)
888  // For actual arrays (not types) this will always be regular whole number.
889  // Types may be variable, fixed, or bounded.
890  IntIndex Length()
891  {
892  // TODO Its going to make sense for dim 0 to
893  // alwasy be the cumulative length, and for 1d arrays that will be the only entry
894  // for 2d there will be cumulateive and individual, with strides for each dim as well
895  // It will only calculated when resized then adn this will work better as an inlined function
896  IntIndex *pDimLength = GetDimensionLengths();

899  while (pDimLength < pEndDimLength) {
900  length *= *pDimLength++;
901  }
902  return length;
903  }
