238  public:
239  bool operator()(const SubString &a, const SubString &b) const
240  {
241  Int32 aSize = a.Length();
242  Int32 bSize = b.Length();
243  if (aSize != bSize) {
244  return aSize < bSize;
245  } else {
246  const Utf8Char* ba = a.Begin();
247  const Utf8Char* bb = b.Begin();
248  const Utf8Char* ea = a.End();
249 
250  while (ba < ea) {
251  if (*ba != *bb) {
252  return *ba < *bb;
253  } else {
254  ba++;
255  bb++;
256  }
257  }
258  return false;
259  }
