251 class ComapreSubString
252 {
253  public:
254  bool operator()(const SubString &a, const SubString &b) const
255  {
256  Int32 aSize = a.Length();
257  Int32 bSize = b.Length();
258  if (aSize != bSize) {
259  return aSize < bSize;
260  } else {
261  const Utf8Char* ba = a.Begin();
262  const Utf8Char* bb = b.Begin();
263  const Utf8Char* ea = a.End();
264 
265  while (ba < ea) {
266  if (*ba != *bb) {
267  return *ba < *bb;
268  } else {
269  ba++;
270  bb++;
271  }
272  }
273  return false;
274  }
275  }
